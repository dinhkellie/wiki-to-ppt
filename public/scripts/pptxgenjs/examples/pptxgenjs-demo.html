Creating Slide Masters is easy!

pptx.defineSlideMaster({ title : 'MASTER_SLIDE', margin: [0.5, 0.25, 1.25, 0.25], bkgd : 'EEEEEE', objects: [ {rect: { x:0, y:6.9, w:'100%', h:0.6, fill:'003b75' }}, {text: { text:'S.T.A.R. Laboratories', options:{x:0.17, y:0.1, color:'FFFFFF'} }} ], slideNumber: { x:1.0, y:7.0, color:'FFFFFF' } });
